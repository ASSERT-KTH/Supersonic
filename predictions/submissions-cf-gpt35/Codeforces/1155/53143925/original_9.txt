typedef long long LL;
const int maxn = 3 * 1e5 + 10;
LL x[maxn], p[maxn], num[maxn];
LL gcd(LL x, LL y) { return y == 0 ? x : gcd(y, x % y); }
int main() {
  int n, m, flag = 0;
  scanf("%d %d", &n, &m);
  for (int i = 1; i <= n; i++) {
    scanf("%lld", &x[i]);
    num[i] = x[i] - x[i - 1];
  }
  for (int i = 1; i <= m; i++) {
    scanf("%lld", &p[i]);
  }
  LL mid = num[2];
  for (int i = 3; i <= n; i++) {
    mid = gcd(mid, num[i]);
  }
  for (int i = 1; i <= m; i++) {
    if (mid % p[i] == 0) {
      printf("YES\n");
      printf("%lld %d\n", x[1], i);
      return 0;
    }
  }

  return 0;
}
