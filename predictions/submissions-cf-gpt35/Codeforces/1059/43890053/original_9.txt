#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <vector>

using namespace std;

const int maxn = 100010;

double x[maxn], y[maxn];
int n;

bool isOk(double k) {
  double l = -1e18, r = 1e18;
  for (int i = 0; i < n; i++) {
    if (y[i] > k * 2) {
      return false;
    }
    double tmp = sqrt(k * k - (k - y[i]) * (k - y[i]));
    l = max(l, x[i] - tmp);
    r = min(r, x[i] + tmp);
  }
  return l < r;
}

int main() {
  scanf("%d", &n);
  for (int i = 0; i < n; i++) {
    scanf("%lf %lf", &x[i], &y[i]);
  }
  for (int i = 0; i < n; i++) {
    if (y[i] * y[n - 1] < 0) {
      printf("-1\n");
      return 0;
    } else {
      y[i] = abs(y[i]);
    }
  }
  double l = 0, r = 1e18;
  double mid = (l + r) / 2.0;
  for (int i = 1; i <= 100; i++) {
    mid = (l + r) / 2.0;
    if (isOk(mid)) {
      r = mid;
    } else {
      l = mid;
    }
  }
  printf("%.10lf\n", mid);
}
