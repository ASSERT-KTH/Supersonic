```c++
#include <bits/stdc++.h>
using namespace std;
int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  int n, m;
  cin >> n >> m;
  unordered_map<string, string> mp;
  for (int i = 0; i < m; i++) {
    string x, y;
    cin >> x >> y;
    if (x.length() <= y.length()) {
      mp[x] = x;
    } else {
      mp[x] = y;
    }
  }
  for (int i = 0; i < n; i++) {
    string z;
    cin >> z;
    cout << mp[z] << " ";
  }
  return 0;
}
```
