```cpp
#include <iostream>
#include <unordered_set>
#include <vector>
#include <cmath>

using namespace std;

void fun() {
  int n, r;
  cin >> n >> r;
  unordered_set<int> f;
  vector<int> a(n);
  for (auto &i : a)
    cin >> i;
  for (int i = 0; i < n; i++) {
    int k = 1;
    for (int j = 2; j * j <= a[i]; j++) {
      int kk = 0;
      while (a[i] % j == 0)
        a[i] /= j, kk++;
      if (kk & 1)
        k *= j;
    }
    if (a[i] != 1)
      k *= a[i];
    a[i] = k;
  }
  int ans = 1;
  for (int i = 0; i < n; i++) {
    if (f.insert(a[i]).second == false)
      ans++, f.clear();
  }
  cout << ans << endl;
}
int main() {
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  int t = 1;
  cin >> t;
  while (t--) {
    fun();
  }
}
```
