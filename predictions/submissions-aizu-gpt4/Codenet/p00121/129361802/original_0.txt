#include <algorithm>
#include <bitset>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
using namespace std;
char bj[8];
bitset<80000000> vis;
struct pt {
  int ztt;
  int lwz;
  int step;
};
int bs;
pt dl[1 << 16];
int front = 1, back = 1;
int tt;
void bfs() {

  front = 1;
  back = 1;
  bs = 0;

  lsss.step = 0;
  dl[back++] = lsss;

    pt ls1 = dl[front++];
    if (ls1.ztt == 1234567) {
      bs = ls1.step;
      tt = 1;
      return;

    int kkk[4] = {-1, 1, -4, 4};

      sss[i] = tt % 10;
      tt /= 10;
    }

          lsjr.step = ls1.step + 1;
          vis[lsjr.ztt] = 1;
          dl[back++] = lsjr;
        }
      }
    }
  }
}
int main() {
  while (scanf("%s", bj) == 1) {
    bfs();

  }

}
