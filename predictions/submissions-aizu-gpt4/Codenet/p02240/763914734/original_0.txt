#include <algorithm>
#include <queue>
#include <stdio.h>
#include <vector>
using namespace std;
class Relation {
public:
  Relation(int tmp_n, int tmp_m) : n(tmp_n), m(tmp_m), group_number(1) {
    V.resize(n);
    num_of_each.resize(n);
    group.resize(n);
  }
  void push(int s, int t) {
    V[s].push_back(t);
    V[t].push_back(s);
    num_of_each[s]++;
    num_of_each[t]++;
  }
  void makeGroup() {
    int node;
    queue<int> Q;
    Q.push(0);
    group[0] = 1;
    while (true) {
      while (!Q.empty()) {
        node = Q.front();
        Q.pop();
        for (const auto &k : V[node]) {
          if (group[k] == 0) {
            group[k] = group[node];
            Q.push(k);
          }
        }
      }
      group_number++;
      auto it = find(group.begin(), group.end(), 0);
      if (it == group.end()) break;
      *it = group_number;
      Q.push(distance(group.begin(), it));
    }
  }
  bool canAccess(int s, int t) { return group[s] == group[t]; }

private:
  vector<vector<int>> V;
  vector<int> num_of_each;
  vector<int> group;
  int group_number;
  int n, m;
};
int main() {
  int n, m, s, t, q;
  scanf("%d %d", &n, &m);
  Relation relation(n, m);
  for (int i = 0; i < m; i++) {
    scanf("%d %d", &s, &t);
    relation.push(s, t);
  }
  relation.makeGroup();
  scanf("%d", &q);
  for (int i = 0; i < q; i++) {
    scanf("%d %d", &s, &t);
    if (relation.canAccess(s, t)) {
      printf("yes\n");
    } else {
      printf("no\n");
    }
  }
  return 0;
}
