#include <vector>
#include <algorithm>
#define r(i, n) for (auto i = 0; i < n; i++)

  std::ios_base::sync_with_stdio(false);
  std::cin.tie(NULL);

  int n;
  std::cin >> n;

  std::vector<int> A(n);
  r(i, n) std::cin >> A[i];

}
