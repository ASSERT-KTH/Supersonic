void matrix(double x[][N + 1]) {
  double su1 = 0;
  for (int k = 0; k < N-1; k++) {
    for (int j = k + 1; j < N; j++) {
      su1 = x[j][k] / x[k][k];
      for (int i = k; i <= N; i++) {
        x[j][i] -= x[k][i] * su1;
      }
    }
  }
  for (int i = N-1; i >= 0; i--) {
    for (int j = i + 1; j < N; j++) {
      x[i][N] -= x[i][j] * x[j][N];
    }
    x[i][N] /= x[i][i];
  }
}
