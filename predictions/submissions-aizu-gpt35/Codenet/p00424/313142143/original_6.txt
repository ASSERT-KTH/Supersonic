#include <iostream>
#include <cstring>
#define z std::cin >>
int main() {
    int n;
    for (char a, b; z n, n; puts("")) {
        int d[256] = {};
        for (; n--; d[a] = b - a)
            z a >> b;
        for (z n; n--; putchar(a + d[a]))
            z a;
        memset(d, 0, sizeof(d));    // reset array for next iteration
    }
    return 0;
}
