  int const MOD = 100000;
  for (int N, M, S; cin >> N >> M >> S && (N | M | S);) {
    zero(dp);
    dp[0] = 1;
    REP(add, 1, M + 1) {
      for (int currSum = S; currSum >= add; currSum--) {
        rep(i, N * N) {
          if (currSum - add*i < add) break;
          dp[currSum] = (dp[currSum] + dp[currSum - add*i]) % MOD;
        }
      }
    }

  }

}
