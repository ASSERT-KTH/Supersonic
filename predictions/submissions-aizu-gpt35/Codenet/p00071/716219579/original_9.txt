void solve(int y, int x) {
  M[y][x] = 0;
  vis[y][x] = true;
  for (int i = 0; i < 4; i++) {
    for (int j = 1; j <= 3; j++) {
      int ny = y + dy[i] * j, nx = x + dx[i] * j;
      if (ny >= 0 && ny < 8 && nx >= 0 && nx < 8 && !vis[ny][nx]) {
        if (M[ny][nx] == 1)
          solve(ny, nx);
        vis[ny][nx] = true;
      }
    }
  }
}

    cin >> x >> y;
    memset(vis, false, sizeof(vis));
    solve(y - 1, x - 1);
